    public static boolean search(int matrix[][],int key)
    {
        int i,j;
        int n=matrix.length,m=matrix[0].length;

            for(j=0;j<m;j++)
            {
                if(matrix[i][j]==key)

                    return true;
                }
            }
        }

        return false;
    }
